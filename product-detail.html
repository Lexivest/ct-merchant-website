import { serve } from "https://deno.land/std@0.168.0/http/server.ts" import { createClient } from "https://esm.sh/@supabase/supabase-js@2" const corsHeaders = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type', } serve(async (req) => { if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders }) try { const supabaseUrl = Deno.env.get('SUPABASE_URL') const supabaseAnonKey = Deno.env.get('SUPABASE_ANON_KEY') const serviceRoleKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') const groqKey = Deno.env.get('GROQ_API_KEY') if (!supabaseUrl || !serviceRoleKey || !groqKey) throw new Error("Missing Server Secrets") // We now expect 'context' from the frontend const { query, history = [], context = {} } = await req.json() const recentHistory = history.slice(-4); const safeHistory = recentHistory.map((h: any) => ({ role: h.role, content: h.content || "" })); const authHeader = req.headers.get('Authorization') let user = null; if (authHeader) { const authClient = createClient(supabaseUrl, supabaseAnonKey, { global: { headers: { Authorization: authHeader } } }) const { data } = await authClient.auth.getUser() if (data && data.user) user = data.user; } const adminClient = createClient(supabaseUrl, serviceRoleKey) const today = new Date().toISOString().split('T')[0]; let finalReplyText = ""; // --- 1. DETERMINE USAGE LIMITS --- let chatCount = 0; let clientIp = 'unknown'; if (user) { const { data: profile } = await adminClient.from('profiles').select('ai_chat_count, ai_last_chat_date').eq('id', user.id).single(); chatCount = profile?.ai_chat_count || 0; let lastDate = profile?.ai_last_chat_date || today; if (lastDate !== today) chatCount = 0; if (chatCount >= 15) { return new Response(JSON.stringify({ reply: "Error: You have reached your daily limit of 15 AI requests. Please check back tomorrow!" }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } } else { clientIp = req.headers.get('x-forwarded-for')?.split(',')[0] || req.headers.get('cf-connecting-ip') || 'unknown'; if (clientIp !== 'unknown') { const { data: usageData } = await adminClient.from('anonymous_ai_usage').select('*').eq('ip_address', clientIp).single(); if (usageData) { chatCount = usageData.chat_count; let anonLastDate = usageData.last_chat_date; if (anonLastDate !== today) chatCount = 0; } if (chatCount >= 15) { return new Response(JSON.stringify({ reply: "Error: You have reached the daily free limit. Please Create a Free Account to continue!" }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } } } // --- 2. BUILD CONTEXT-AWARE SYSTEM PROMPT --- let systemMessage = ""; let tools = []; // If the user is on the Product Detail page if (context.page === 'product_detail' && context.product) { const p = context.product; const s = context.shop || {}; systemMessage = `You are the CTMerchant AI Concierge. CURRENT CONTEXT: The user is looking at a specific product. Product Name: ${p.name} Price: ₦${p.discount_price || p.price} Condition: ${p.condition} Description: ${p.description} Shop Name: ${s.name || 'Unknown'} Shop Address: ${s.address || 'Unknown'} STRICT DIRECTIVES: 1. STAY ON TOPIC: Focus ONLY on this product. If asked about unrelated things, politely direct them to use the homepage search bar. 2. SECURITY (CRITICAL): Strongly advise the user to VISIT THE PHYSICAL SHOP (${s.address}) to inspect the item. Explicitly WARN them NEVER to transfer money to a seller's bank account before seeing the product in person. 3. CONTACT: Tell them to use the 'WhatsApp' or 'Call Seller' buttons at the bottom of the screen to negotiate or ask the seller questions. 4. PROACTIVE OFFER: Ask: "Would you like me to find similar items in this category to compare prices?" 5. HTML FORMATTING: If you use the 'find_similar_products' tool, format the results exactly like this:
[Product Name]
₦[Price] - Sold by [Shop Name]
View Product
`; tools = [{ type: "function", function: { name: "find_similar_products", description: "Finds other products in the same category to compare prices.", parameters: { type: "object", properties: {}, required: [] } } }]; } // Generic Fallback Prompt else { systemMessage = `You are CTMerchant's AI Assistant. "CTMerchant is a verified repository of physical shops." Keep responses concise, friendly, and human-like. Do NOT attempt to search for products.`; } const messages = [{ role: "system", content: systemMessage }, ...safeHistory, { role: "user", content: query }]; // --- 3. CALL GROQ (LLAMA 3) --- const groqCall1 = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${groqKey}` }, body: JSON.stringify({ model: "llama-3.3-70b-versatile", messages: messages, tools: tools.length > 0 ? tools : undefined, temperature: 0.2, max_tokens: 300 }) }); const data1 = await groqCall1.json(); if (!data1.choices || data1.choices.length === 0) throw new Error("Groq API returned an empty response."); const responseMessage = data1.choices[0].message; // --- 4. HANDLE TOOL CALLING (Product Comparison) --- if (responseMessage.tool_calls) { messages.push(responseMessage); for (const toolCall of responseMessage.tool_calls) { if (toolCall.function.name === "find_similar_products") { let toolResultString = ""; // Fetch similar products natively using Supabase const { data: similarProducts, error: searchErr } = await adminClient .from('products') .select('id, name, price, discount_price, shops(name)') .eq('category', context.product.category) .eq('is_available', true) .neq('id', context.product.id) .limit(3); if (searchErr) { toolResultString = `Database Error: ${searchErr.message}`; } else if (similarProducts && similarProducts.length > 0) { toolResultString = similarProducts.map((p: any) => { const price = p.discount_price || p.price; return `ID: ${p.id} | Name: ${p.name} | Price: ${price} | Shop: ${p.shops?.name || 'Unknown'}`; }).join('\n'); } else { toolResultString = `No similar products found in this category right now.`; } messages.push({ tool_call_id: toolCall.id, role: "tool", name: toolCall.function.name, content: toolResultString }); } } // Second call with tool results const groqCall2 = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${groqKey}` }, body: JSON.stringify({ model: "llama-3.3-70b-versatile", messages: messages, temperature: 0.2, max_tokens: 400 }) }); const data2 = await groqCall2.json(); finalReplyText = data2.choices[0].message.content; } else { finalReplyText = responseMessage.content; } // --- 5. UPDATE USAGE LIMITS --- if (user) { await adminClient.from('profiles').update({ ai_chat_count: chatCount + 1, ai_last_chat_date: today }).eq('id', user.id); } else if (clientIp !== 'unknown') { await adminClient.from('anonymous_ai_usage').upsert({ ip_address: clientIp, chat_count: chatCount + 1, last_chat_date: today }); } return new Response(JSON.stringify({ reply: finalReplyText }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } catch (error) { return new Response(JSON.stringify({ reply: `System Error: ${error.message}` }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } })
